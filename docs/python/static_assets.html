 178  function distribute(n, width, top, scale) {
 179    const slots = [];
 180    if (n <= 0) return slots;
 181    if (n === 1) {
 182      slots.push({ left: 50, top, scale });
 183      return slots;
 184    }
 185
 186    const margin = (100 - width) / 2;
 187    const step = width / (n - 1);
 188    for (let i = 0; i < n; i++) {
 189      slots.push({ left: margin + i * step, top, scale });
 190    }

 400     Text paging
 401     ------------------------------------------------------------------ */
 402  function splitIntoPages(text) {
 403    const style = getComputedStyle($text);
 404    const maxHeight = parseFloat(style.maxHeight);
 405    if (!maxHeight || !text) return [text];
 406
 407    const measurer = document.createElement('div');
 408    measurer.style.position = 'absolute';
 409    measurer.style.visibility = 'hidden';
 410    measurer.style.left = '-9999px';
 411    measurer.style.width = $text.clientWidth + 'px';
 412    measurer.style.font = style.font;
 413    measurer.style.lineHeight = style.lineHeight;
 414    measurer.style.whiteSpace = 'pre-wrap';
 415    measurer.style.wordBreak = 'break-word';
 416    document.body.appendChild(measurer);
 417
 418    const pages = [];
 419    let remaining = text;
 420
 421    while (remaining.length > 0) {
 422      let low = 1, high = remaining.length, best = 1;
 423      while (low <= high) {
 424        const mid = Math.floor((low + high) / 2);
 425        measurer.textContent = remaining.slice(0, mid);
 426        if (measurer.offsetHeight <= maxHeight) {
 427          best = mid;
 428          low = mid + 1;
 429        } else {
 430          high = mid - 1;
 431        }
 432      }
 433
 434      // Prefer breaking at a word boundary
 435      let split = best;
 436      if (split < remaining.length) {
 437        const before = remaining.lastIndexOf(' ', split - 1);
 438        if (before > 0 && split - before < 25) {
 439          split = before + 1;
 440        }
 441      }
 442      if (split < 1) split = 1;
 443
 444      pages.push(remaining.slice(0, split));
 445      remaining = remaining.slice(split);
 446    }
 447
 448    document.body.removeChild(measurer);
 449    return pages.length ? pages : [text];
 450  }

 498      }

 651    }

 654      applyEnterExit(ev.enter, ev.exit);

 656        Object.entries(ev.sprite_changes).forEach(([name, sprite]) => {
 657          const char = STATE.here.find(c => c.name === name);

 665      updateSprites(STATE.here, ev.speaker || null);
 666
 667      const output = ev.output || '';
 668      if (!output) return 'continue';
 669
 670      addToHistory(ev.speaker || null, output);
 671      await typeText(ev.speaker || null, output);
 672      return 'wait';
 673    }
 674
 675    if (ev.type === 'needs_player_input') {
 676      applyEnterExit(ev.enter, ev.exit);
 677      if (ev.sprite_changes) {
 678        Object.entries(ev.sprite_changes).forEach(([name, sprite]) => {
 679          const char = STATE.here.find(c => c.name === name);
 680          if (char) char.current_sprite = sprite;
 681        });
 682      }
 683      if (ev.system_changes && window.applySystemChanges) {
 684        window.applySystemChanges(ev.system_changes);
 685      }

 687      updateSprites(STATE.here, ev.speaker || null);
 688      showChoices(ev.suggestions || []);

 693      clearSpeaker();
 694      $text.textContent = '— Scene End —';
 695      hideChoices();

 697        setBackground(ev.loading_background);
 698      }

 701    }
 702
 703    if (ev.type === 'scene_ended') {
 704      clearSpeaker();
 705      $text.textContent = '— Scene End —';
 706      hideChoices();
 707      if (ev.loading_background) {
 708        setBackground(ev.loading_background);
 709      }
 710      showTransition(ev.next_scene, ev.next_scene_name);
 711      return 'continue';
 712    }
 713
 714    if (ev.type === 'story_complete') {
 715      clearSpeaker();
 716      $text.textContent = '— The End —';
 717      hideChoices();
 718      await hideTransition();
 719      return 'done';
 720    }
 721
 722    return 'continue';
 723  }

 728  function waitForClick() {
 729    return new Promise(resolve => {
 730      let autoTimer = null;
 731      let pollInterval = null;
 732      let finished = false;
 733      _waitResolve = resolve;
 734
 735      function finish() {
 736        if (finished) return;
 737        finished = true;
 738        _waitResolve = null;
 739        if (autoTimer) {
 740          clearTimeout(autoTimer);
 741          autoTimer = null;
 742        }
 743        if (pollInterval) {
 744          clearInterval(pollInterval);
 745          pollInterval = null;
 746        }
 747        cleanup();
 748        resolve();
 749      }
 750
 751      function onClick(e) {
 752        if (e.target.closest('.vn-choice') ||
 753            e.target.closest('.vn-title-btn') ||
 754            e.target.closest('.vn-control-btn') ||
 755            e.target.closest('#vn-controls') ||
 756            e.target.closest('#vn-settings') ||
 757            e.target.closest('#vn-history') ||
 758            e.target.closest('#vn-keybinds') ||
 759            e.target.closest('#vn-saveload') ||
 760            e.target.closest('#vn-debug') ||
 761            e.target.closest('#vn-system')) {
 762          return;
 763        }
 764        if (isTyping()) {
 765          skipTyping();
 766          return;
 767        }
 768        finish();
 769      }
 770      function onKey(e) {
 771        if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA' || e.target.isContentEditable) {
 772          return;
 773        }
 774        if (e.code === 'Space') {
 775          e.preventDefault();
 776          if (isTyping()) {
 777            skipTyping();
 778            return;
 779          }
 780          finish();
 781        }
 782      }
 783      function cleanup() {
 784        document.removeEventListener('click', onClick);
 785        document.removeEventListener('keydown', onKey);
 786      }
 787      document.addEventListener('click', onClick);
 788      document.addEventListener('keydown', onKey);
 789
 790      function maybeStartAuto() {
 791        if (finished) return;
 792        if (_autoMode && !autoTimer) {
 793          autoTimer = setTimeout(finish, STATE.autoDelay);
 794        } else if (!_autoMode && autoTimer) {
 795          clearTimeout(autoTimer);
 796          autoTimer = null;
 797        }
 798      }
 799
 800      maybeStartAuto();
 801      pollInterval = setInterval(maybeStartAuto, 200);
 802    });
 803  }

 806     Main loop
 807     ------------------------------------------------------------------ */
 808  async function gameLoop() {
 809    if (_loopActive) return;
 810    _loopActive = true;
 811    while (_running) {
 812      try {
 813        // Loading indicator while waiting for the server.
 814        if (!_transitioning) {
 815          clearSpeaker();
 816          $text.innerHTML = '<span style="color:var(--vn-gold)">…</span>';
 817        }
 818
 819        _abortController = new AbortController();
 820        const resp = await fetch('/step', {
 821          method: 'POST',
 822          signal: _abortController.signal,
 823        });
 824        _abortController = null;
 825        if (!resp.ok) {
 826          console.error('/step failed', await resp.text());
 827          await sleep(1000);
 828          continue;
 829        }
 830        const ev = await resp.json();
 831        console.log('Step event:', ev);
 832
 833        const result = await processEvent(ev);
 834        if (result === 'wait') {
 835          await waitForClick();
 836        } else if (result === 'input') {
 837          break;
 838        } else if (result === 'done') {
 839          _running = false;
 840          break;
 841        }
 842      } catch (err) {
 843        if (err.name === 'AbortError') {
 844          // Fetch was cancelled by load/start/reset – exit cleanly.
 845          break;
 846        }
 847        console.error('Game loop error', err);
 848        await sleep(1000);
 849      }
 850    }
 851    _loopActive = false;
 852  }
